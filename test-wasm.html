The Inline WASM Byte Array

// WebAssembly module as raw bytes
// Exports: add(i32, i32) -> i32

const wasmBytes = new Uint8Array([
  0x00, 0x61, 0x73, 0x6D,  // magic: "\0asm"
  0x01, 0x00, 0x00, 0x00,  // version: 1

  // Type section
  0x01, 0x07,              // section id=1, size=7
  0x01,                    // 1 type
  0x60,                    // func type
  0x02, 0x7F, 0x7F,        // 2 params: i32, i32
  0x01, 0x7F,              // 1 result: i32

  // Function section
  0x03, 0x02,              // section id=3, size=2
  0x01, 0x00,              // 1 func using type 0

  // Export section
  0x07, 0x07,              // section id=7, size=7
  0x01,                    // 1 export
  0x03, 0x61, 0x64, 0x64,  // name: "add"
  0x00, 0x00,              // func, index 0

  // Code section
  0x0A, 0x09,              // section id=10, size=9
  0x01, 0x07, 0x00,        // 1 body, size=7, 0 locals
  0x20, 0x00,              // local.get 0
  0x20, 0x01,              // local.get 1
  0x6A,                    // i32.add
  0x0B                     // end
]);
